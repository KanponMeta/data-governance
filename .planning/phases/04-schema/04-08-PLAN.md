<objective>
Phase 4 以三个 CLI 子命令、EXPLAIN ANALYZE 人工验证工具和 E2E 测试套件作为收尾，通过 CLI 和 REST 表面端到端地演练每个 ROADMAP 验收标准。

具体而言，本计划交付：

1. **CLI 子命令** — `./platform impact`、`./platform schema ack-break`、`./platform schema diff`、`./platform lineage export`。所有四个都是面向操作员的工具，封装了 Wave 7 REST handler 使用的相同库函数（D-19 的"三个表面，一个逻辑"：Go 库 + REST + CLI）。操作员今天就有了一个调查工具；外部告警/变更管理机器人使用 REST。
2. **EXPLAIN ANALYZE 工具**（D-14, VALIDATION.md 人工验证）— 一个可重复的 shell 脚本，将 10K 条边 seed 到测试数据库，在深度 10 和 25 处运行递归 CTE，并捕获查询计划。捕获的计划提交到 `.planning/phases/04-schema/04-EXPLAIN.md` 并由阶段验证报告引用。这是 VALIDATION.md 指出的唯一人工执行的步骤。
3. **Phase 4 E2E 测试套件** — 一个位于 `test/integration/phase4_e2e_test.go` 的 Go 测试文件，端到端地演练所有 5 个 ROADMAP 验收标准：物化后的 lineage 捕获、对 5 资产链的影响分析、带 breaking change 的 schema diff、metadata PATCH 生效、OpenLineage 导出形状合规。
4. **最终 main.go 子命令布线** — 在 `cmd/platform/main.go` 的调度 switch 中添加 `impact`、`schema` 和 `lineage` 分支。现有 Phase 1-3 分支（`start`、`migrate`、`worker`、`materialize`、`scheduler`、`backfill`）不受影响。

       package main

       import (
           "context"
           "encoding/json"
           "errors"
           "flag"
           "fmt"
           "os"
           "text/tabwriter"
           "time"

           "github.com/kanpon/data-governance/internal/config"
           "github.com/kanpon/data-governance/internal/lineage/impact"
           lineageq "github.com/kanpon/data-governance/internal/lineage/queries"
           "github.com/kanpon/data-governance/internal/storage"
       )

       func runImpact(args []string) error {
           fs := flag.NewFlagSet("impact", flag.ContinueOnError)
           direction := fs.String("direction", "downstream", "upstream or downstream")
           depth := fs.Int("depth", impact.DefaultDepth, "traversal depth (max 25)")
           column := fs.String("column", "", "optional column-level traversal")
           asOfStr := fs.String("as-of", "", "RFC3339 timestamp for point-in-time query")
           format := fs.String("format", "table", "table or json")
           if err := fs.Parse(args); err != nil { return err }
           if fs.NArg() < 1 {
               return fmt.Errorf("usage: ./platform impact <asset> [--direction=down|up] [--depth=N] [--column=COL] [--as-of=...] [--format=table|json]")
           }
           asset := fs.Arg(0)

           cfg, err := config.Load()
           if err != nil { return err }
           ctx, cancel := context.WithTimeout(context.Background(), 30*time.Second)
           defer cancel()
           store, err := storage.NewPostgres(ctx, cfg.DatabaseURL)
           if err != nil { return err }
           defer store.Close()

           q := impact.ImpactQuery{Asset: asset, Direction: *direction, Depth: *depth}
           if *column != "" { q.Column = column }
           if *asOfStr != "" {
               t, err := time.Parse(time.RFC3339, *asOfStr)
               if err != nil { return fmt.Errorf("--as-of must be RFC3339: %w", err) }
               q.AsOf = &t
           }
           result, err := impact.Analyze(ctx, lineageq.New(store.DB()), q)
           if err != nil {
               if errors.Is(err, impact.ErrDepthExceeded) {
                   return fmt.Errorf("depth %d exceeds hard cap of 25 (D-14)", *depth)
               }
               return err
           }

           switch *format {
           case "json":
               return json.NewEncoder(os.Stdout).Encode(result)
           case "table":
               tw := tabwriter.NewWriter(os.Stdout, 0, 0, 2, ' ', 0)
               fmt.Fprintln(tw, "DEPTH\tASSET\tCOLUMN")
               for _, n := range result.Nodes {
                   col := ""
                   if n.Column != nil { col = *n.Column }
                   fmt.Fprintf(tw, "%d\t%s\t%s\n", n.Depth, n.Asset, col)
               }
               return tw.Flush()
           default:
               return fmt.Errorf("--format must be table or json (got %q)", *format)
           }
       }

       package main

       import (
           "context"
           "encoding/json"
           "flag"
           "fmt"
           "os"
           "text/tabwriter"
           "time"

           "github.com/google/uuid"
           "github.com/kanpon/data-governance/internal/config"
           "github.com/kanpon/data-governance/internal/connector"
           "github.com/kanpon/data-governance/internal/event"
           "github.com/kanpon/data-governance/internal/schema"
           "github.com/kanpon/data-governance/internal/storage"
           ent "github.com/kanpon/data-governance/internal/storage/ent"
           schemachange "github.com/kanpon/data-governance/internal/storage/ent/schemachange"
           schemaversion "github.com/kanpon/data-governance/internal/storage/ent/schemaversion"
       )

       // dispatchSchema routes ./platform schema <subcommand>
       func dispatchSchema(args []string) error {
           if len(args) == 0 {
               return fmt.Errorf("usage: ./platform schema {ack-break|diff} ...")
           }
           switch args[0] {
           case "ack-break":
               return runSchemaAckBreak(args[1:])
           case "diff":
               return runSchemaDiff(args[1:])
           default:
               return fmt.Errorf("unknown schema subcommand: %s (try ack-break or diff)", args[0])
           }
       }

       func runSchemaAckBreak(args []string) error {
           fs := flag.NewFlagSet("schema ack-break", flag.ContinueOnError)
           reason := fs.String("reason", "", "REQUIRED — free-text reason for acknowledging this break")
           actorIDStr := fs.String("actor", "", "REQUIRED — actor user UUID (the operator running the ack)")
           if err := fs.Parse(args); err != nil { return err }
           if fs.NArg() < 2 {
               return fmt.Errorf("usage: ./platform schema ack-break <asset> <change_id> --reason=\"...\" --actor=<user_uuid>")
           }
           if *reason == "" { return fmt.Errorf("--reason is required (D-10: no silent acks)") }
           if *actorIDStr == "" { return fmt.Errorf("--actor (UUID) is required") }
           actor, err := uuid.Parse(*actorIDStr)
           if err != nil { return fmt.Errorf("--actor must be UUID: %w", err) }
           changeID, err := uuid.Parse(fs.Arg(1))
           if err != nil { return fmt.Errorf("change_id must be UUID: %w", err) }
           // assetArg := fs.Arg(0) // recorded for human readability; uniqueness is via change_id

           cfg, err := config.Load()
           if err != nil { return err }
           ctx, cancel := context.WithTimeout(context.Background(), 30*time.Second)
           defer cancel()
           store, err := storage.NewPostgres(ctx, cfg.DatabaseURL)
           if err != nil { return err }
           defer store.Close()
           writer := event.NewWriter(store)
           entClient, err := ent.Open("postgres", cfg.DatabaseURL)
           if err != nil { return err }
           defer entClient.Close()

           existing, err := entClient.SchemaChange.Query().
               Where(schemachange.IDEQ(changeID)).Only(ctx)
           if err != nil { return fmt.Errorf("schema_change not found: %w", err) }
           if existing.AcknowledgedAt != nil {
               return fmt.Errorf("schema_change %s was already acknowledged at %s", changeID, existing.AcknowledgedAt)
           }
           now := time.Now().UTC()
           updated, err := entClient.SchemaChange.UpdateOneID(changeID).
               SetAcknowledgedAt(now).
               SetAcknowledgedBy(actor).
               SetAcknowledgementReason(*reason).
               Save(ctx)
           if err != nil { return err }
           _ = writer.Append(ctx, event.Event{
               Type: event.EventTypeSchemaBreakAcknowledged,
               OccurredAt: now,
               ActorID: &actor,
               ResourceType: "schema_change",
               ResourceID: changeID.String(),
               Payload: map[string]any{
                   "schema_change_id": changeID.String(),
                   "asset": existing.Asset,
                   "reason": *reason,
               },
           })
           fmt.Printf("acknowledged schema_change %s for asset %s at %s by %s\n",
               changeID, existing.Asset, now.Format(time.RFC3339), actor)
           _ = updated
           return nil
       }

       func runSchemaDiff(args []string) error {
           fs := flag.NewFlagSet("schema diff", flag.ContinueOnError)
           assetName := fs.String("asset", "", "asset name (required)")
           fromVer := fs.String("from", "", "schema_version UUID (required)")
           toVer := fs.String("to", "", "schema_version UUID (required)")
           format := fs.String("format", "table", "table or json")
           if err := fs.Parse(args); err != nil { return err }
           if *assetName == "" || *fromVer == "" || *toVer == "" {
               return fmt.Errorf("usage: ./platform schema diff --asset=NAME --from=UUID --to=UUID [--format=table|json]")
           }
           fromUUID, err := uuid.Parse(*fromVer)
           if err != nil { return err }
           toUUID, err := uuid.Parse(*toVer)
           if err != nil { return err }

           cfg, err := config.Load()
           if err != nil { return err }
           ctx, cancel := context.WithTimeout(context.Background(), 30*time.Second)
           defer cancel()
           entClient, err := ent.Open("postgres", cfg.DatabaseURL)
           if err != nil { return err }
           defer entClient.Close()

           prevRow, err := entClient.SchemaVersion.Query().Where(schemaversion.IDEQ(fromUUID)).Only(ctx)
           if err != nil { return fmt.Errorf("from schema_version not found: %w", err) }
           nextRow, err := entClient.SchemaVersion.Query().Where(schemaversion.IDEQ(toUUID)).Only(ctx)
           if err != nil { return fmt.Errorf("to schema_version not found: %w", err) }

           // Wave 4 plan 04-04 stores SchemaVersion.SchemaData as JSON-encoded connector.Schema
           // (json.Marshal at write time per the schema.Writer.Capture INSERT).
           // Wave 5 plan 04-05 produced schema.Diff(prev, next connector.Schema) []SchemaChange
           // and schema.Classify(c, lattice) (string, bool) — we use both here.
           prevSchema, err := unmarshalSchema(prevRow.SchemaData)
           if err != nil { return fmt.Errorf("decode prev schema: %w", err) }
           nextSchema, err := unmarshalSchema(nextRow.SchemaData)
           if err != nil { return fmt.Errorf("decode next schema: %w", err) }
           changes := schema.Diff(prevSchema, nextSchema)

           switch *format {
           case "json":
               // Enrich each change with classification before emitting (lattice = schema.IsWideningPostgres).
               type classifiedChange struct {
                   schema.SchemaChange
                   ChangeType string `json:"change_type"`
                   IsBreaking bool   `json:"is_breaking"`
               }
               out := make([]classifiedChange, 0, len(changes))
               for _, c := range changes {
                   ct, breaking := schema.Classify(c, schema.IsWideningPostgres)
                   out = append(out, classifiedChange{SchemaChange: c, ChangeType: ct, IsBreaking: breaking})
               }
               return json.NewEncoder(os.Stdout).Encode(out)
           case "table":
               tw := tabwriter.NewWriter(os.Stdout, 0, 0, 2, ' ', 0)
               fmt.Fprintln(tw, "KIND\tCOLUMN\tBREAKING\tPREV\tNEXT")
               for _, c := range changes {
                   _, breaking := schema.Classify(c, schema.IsWideningPostgres)
                   prev := ""
                   if c.PrevType != nil { prev = *c.PrevType }
                   next := ""
                   if c.NewType != nil { next = *c.NewType }
                   // SchemaChange.ColumnName is plain string per plan 04-05; "" for PK-level changes.
                   fmt.Fprintf(tw, "%v\t%s\t%v\t%s\t%s\n", c.Kind, c.ColumnName, breaking, prev, next)
               }
               return tw.Flush()
           default:
               return fmt.Errorf("--format must be table or json")
           }
       }

       // unmarshalSchema decodes the SchemaVersion.SchemaData JSONB column into a connector.Schema.
       // Wave 4 plan 04-04 stores it via json.Marshal(connector.Schema) in schema.Writer.Capture.
       // The ent field type is json.RawMessage (JSONB column → Go []byte / json.RawMessage).
       func unmarshalSchema(raw json.RawMessage) (connector.Schema, error) {
           var s connector.Schema
           if err := json.Unmarshal(raw, &s); err != nil {
               return connector.Schema{}, fmt.Errorf("unmarshal schema_data: %w", err)
           }
           return s, nil
       }
