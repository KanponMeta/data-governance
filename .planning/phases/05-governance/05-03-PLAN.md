  <behavior>
    - **Algorithm:** for each output column declared on this run's asset:
      1. If `asset_metadata` row for `(asset, column)` carries `pii_override` tag with non-empty `pii_override_reason` → SKIP (override wins).
      2. Otherwise: SELECT EXISTS (any upstream column referenced by `column_edges` with `superseded_at IS NULL` AND `to_asset=$asset, to_column=$col` AND its source's `asset_metadata.tags ? 'pii'`).
      3. If true: UPSERT `asset_metadata.tags = tags || '{"pii":true,"pii_propagated_from": "<list>"}'::jsonb`.
      4. Each new pii=true assignment emits NO new audit entry (propagation is a side-effect of normal materialization). The COUNTERFACTUAL — TagOverride that REMOVES pii — emits `metadata.tag_overridden`.
    - **Override** stored in `asset_metadata.tags` as `{"pii_override": true, "pii_override_reason": "...", "pii_override_actor": "<uuid>", "pii_override_at": "<ts>"}`. Override declared at builder time; loaded into asset_metadata at registration. Override is also accepted at runtime via REST PATCH (deferred — current task ships builder DSL only; REST is plan 06 surface).
    - **Builder DSL** chainable on `*ColumnBuilder` (Phase 4 already exposes `Column(name) *ColumnBuilder`):
      ```go
      asset.New("orders_anonymized").
          Column("hashed_ssn").
              TagOverride(asset.TagOverride{Remove: "pii", Reason: "hashed at source via D-03 MaskHash; not reversible"})
      ```
    - **TagOverride struct**: `{Remove string; Add string; Reason string}`. Either Remove or Add MUST be non-empty; Reason is ALWAYS required and non-empty.
    - **Hook in lineage.Writer.CaptureRun**: after `column_edges` UPSERT, before tx commit, call `propagator.Propagate(ctx, tx, runID, outputColumnRefs)`.
    - **Audit on override application** — when an override exists for the first time on this asset_version (from builder declaration), `Propagate` writes `metadata.tag_overridden` to audit_log with `{actor: 0-uuid (system), asset, column, removed_tag, reason}`. Subsequent reruns with same code_hash do NOT re-emit the audit (idempotent — checked via `asset_metadata.pii_override_audit_seq` field).
    - **Same-tx guarantee:** propagator MUST run inside the caller's `*sql.Tx` (the same one that just wrote column_edges). It MUST NOT open a new connection or new transaction.

```go
import (
    "context"
    "database/sql"
    "encoding/json"
    "time"

    "github.com/google/uuid"
    "github.com/kanpon/data-governance/internal/audit"
)

type ColumnRef struct { Asset, Column string }
type Propagator struct{}
func NewPropagator() *Propagator { return &Propagator{} }

// Propagate is invoked by lineage.Writer.CaptureRun in the same tx.
// It (1) propagates pii=true via union over column_edges,
// (2) emits metadata.tag_overridden audit entries the first time an override
//     is observed for an asset_version (idempotent via pii_override_audit_seq).
func (p *Propagator) Propagate(ctx context.Context, tx *sql.Tx, runID uuid.UUID, outputColumns []ColumnRef, overrides []ColumnTagOverride) error
```

Body:
```go
overrideSet := map[string]ColumnTagOverride{}
for _, o := range overrides {
    overrideSet[o.Asset+"."+o.Column] = o
}

for _, c := range outputColumns {
    key := c.Asset+"."+c.Column

    // 1. Apply override if present
    if o, ok := overrideSet[key]; ok {
        // (a) Read asset_metadata.tags + pii_override_audit_seq
        var existingTags []byte
        var auditSeq sql.NullInt64
        err := tx.QueryRowContext(ctx,
            `SELECT tags, pii_override_audit_seq FROM asset_metadata WHERE asset=$1 AND column_name=$2`,
            c.Asset, c.Column,
        ).Scan(&existingTags, &auditSeq)
        if err != nil && err != sql.ErrNoRows { return err }

        // (b) Compute new tags JSON: remove o.Override.Remove, add o.Override.Add, set pii_override marker
        newTags := mergeOverrideTags(existingTags, o.Override)

        // (c) UPSERT
        if _, err := tx.ExecContext(ctx, `
            INSERT INTO asset_metadata (asset, column_name, tags, set_by, set_at, pii_override_audit_seq)
            VALUES ($1, $2, $3::jsonb, $4, NOW(), $5)
            ON CONFLICT (asset, column_name) DO UPDATE
              SET tags = EXCLUDED.tags,
                  set_at = NOW(),
                  pii_override_audit_seq = COALESCE(asset_metadata.pii_override_audit_seq, EXCLUDED.pii_override_audit_seq)`,
            c.Asset, c.Column, newTags, uuid.Nil, sql.NullInt64{Valid:false},
        ); err != nil { return err }

        // (d) First-time override on this asset_version → emit audit
        if !auditSeq.Valid {
            payload := map[string]any{
                "asset": c.Asset, "column": c.Column,
                "removed_tag": o.Override.Remove,
                "added_tag":   o.Override.Add,
                "reason":      o.Override.Reason,
                "run_id":      runID.String(),
            }
            seq, err := audit.WriteEntry(ctx, tx, audit.Entry{
                EventType:    audit.AuditMetadataTagOverridden,
                OccurredAt:   time.Now().UTC(),
                ActorID:      nil,  // system (builder declaration); runtime PATCH would set actor
                ResourceType: "column",
                ResourceID:   c.Asset+"."+c.Column,
                Payload:      payload,
            })
            if err != nil { return err }
            // Update asset_metadata to record the audit seq so re-runs are idempotent
            if _, err := tx.ExecContext(ctx, `
                UPDATE asset_metadata SET pii_override_audit_seq=$1 WHERE asset=$2 AND column_name=$3`,
                seq, c.Asset, c.Column); err != nil { return err }
        }

        continue  // override wins; skip propagation
    }

    // 2. No override — propagate
    var anyUpstreamPII bool
    if err := tx.QueryRowContext(ctx, `
        SELECT EXISTS (
            SELECT 1 FROM column_edges ce
            JOIN asset_metadata am ON am.asset = ce.from_asset AND am.column_name = ce.from_column
            WHERE ce.to_asset = $1 AND ce.to_column = $2
              AND ce.superseded_at IS NULL
              AND am.tags ? 'pii'
        )`,
        c.Asset, c.Column,
    ).Scan(&anyUpstreamPII); err != nil {
        return err
    }

    if anyUpstreamPII {
        if _, err := tx.ExecContext(ctx, `
            INSERT INTO asset_metadata (asset, column_name, tags, set_by, set_at)
            VALUES ($1, $2, '{"pii":true,"pii_propagated":true}'::jsonb, $3, NOW())
            ON CONFLICT (asset, column_name) DO UPDATE
              SET tags = asset_metadata.tags || '{"pii":true,"pii_propagated":true}'::jsonb,
                  set_at = NOW()`,
            c.Asset, c.Column, uuid.Nil,
        ); err != nil {
            return err
        }
    }
}
return nil
```

    - **Mask functions** (pure Go, deterministic):
      - `MaskHash(value, salt string) string` returns `hex(HMAC-SHA256(salt, value))`. Salt loaded from `MASK_HASH_SALT` env var; production mode (env var `GOV_ENV=prod`) requires non-empty salt or `Apply` returns ErrMaskSaltMissing.

```go
package policy

import (
    "crypto/hmac"
    "crypto/sha256"
    "encoding/hex"
    "errors"
    "os"
    "strings"
    "sync"

    "github.com/kanpon/data-governance/internal/connector"
)

var ErrMaskSaltMissing = errors.New("policy: MASK_HASH_SALT env var is empty in prod mode")

var saltOnce sync.Once
var cachedSalt string
var saltErr error

func Salt() (string, error) {
    saltOnce.Do(func() {
        cachedSalt = os.Getenv("MASK_HASH_SALT")
        if cachedSalt == "" && os.Getenv("GOV_ENV") == "prod" {
            saltErr = ErrMaskSaltMissing
        }
    })
    return cachedSalt, saltErr
}

func ApplyHash(value string) (string, error) {
    salt, err := Salt()
    if err != nil { return "", err }
    h := hmac.New(sha256.New, []byte(salt))
    h.Write([]byte(value))
    return hex.EncodeToString(h.Sum(nil)), nil
}
func ApplyRedact(_ string) string { return "***" }
func ApplyPartial(value string, reveal int) string {
    if reveal <= 0 { reveal = 2 }
    if len(value) <= 2*reveal+1 { return "***" }
    return value[:reveal] + strings.Repeat("*", len(value)-2*reveal) + value[len(value)-reveal:]
}

func Apply(mt connector.MaskType, value string, reveal int) (string, error) {
    switch mt {
    case connector.MaskHash:    return ApplyHash(value)
    case connector.MaskRedact:  return ApplyRedact(value), nil
    case connector.MaskPartial: return ApplyPartial(value, reveal), nil
    default: return value, nil
    }
}

func DefaultMaskForPII() connector.MaskType { return connector.MaskRedact }
```
