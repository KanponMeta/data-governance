func (c *Connector) QueryAggregate(ctx context.Context, ref connector.AssetRef, sqlText string) (connector.Row, error) {
    // c.db is *sql.DB / *pgxpool.Pool from Phase 2
    rows, err := c.db.QueryContext(ctx, sqlText)
    if err != nil { return connector.Row{}, fmt.Errorf("postgres.QueryAggregate: exec: %w", err) }
    defer rows.Close()
    cols, _ := rows.Columns()
    if !rows.Next() { return connector.Row{}, errors.New("no rows") }
    vals := make([]any, len(cols))
    ptrs := make([]any, len(cols))
    for i := range vals { ptrs[i] = &vals[i] }
    if err := rows.Scan(ptrs...); err != nil { return connector.Row{}, err }
    return connector.Row{Columns: cols, Values: vals}, nil
}
```

    row, err := eval.QueryAggregate(ctx, sqlText)
    if err != nil { return QualityResult{Status:"error", ErrorMessage: err.Error()}, nil }
    total, _ := row.Values[0].(float64)
    nulls, _ := row.Values[1].(float64)
    rate := 0.0
    if total > 0 { rate = nulls / total }
    if rate > n.MaxRate {
        return QualityResult{Status:"failed", MeasuredValue: &rate, Threshold: &n.MaxRate}, nil
    }
    return QualityResult{Status:"passed", MeasuredValue: &rate, Threshold: &n.MaxRate}, nil
}
```

    Secret   []byte
    Timeout  time.Duration   // default 30s

func (w *WebhookChannel) Send(ctx context.Context, p SendPayload) error {
    body := p.Body  // pre-rendered JSON
    ts := strconv.FormatInt(p.Timestamp.Unix(), 10)
    sig := computeHMAC(w.Secret, []byte(ts+"."+string(body)))
    req, _ := http.NewRequestWithContext(ctx, "POST", w.URL, bytes.NewReader(body))
    req.Header.Set("Content-Type", "application/json")
    req.Header.Set("X-Platform-Webhook-ID", p.WebhookID)
    req.Header.Set("X-Platform-Timestamp", ts)
    req.Header.Set("X-Platform-Signature", sig)
    resp, err := w.Client.Do(req)
    if err != nil { return err }
    defer resp.Body.Close()
    if resp.StatusCode >= 300 { return fmt.Errorf("webhook %d", resp.StatusCode) }
    return nil
}
func computeHMAC(secret, msg []byte) string {
    h := hmac.New(sha256.New, secret); h.Write(msg); return hex.EncodeToString(h.Sum(nil))
}

    Router *Router
    Events event.Writer
    DB     *sql.DB    // for emitting notification.dispatched event_log

func (w *Worker) Work(ctx context.Context, job *river.Job[NotificationDispatchArgs]) error {
    channels := w.Router.Route(ctx, job.Args.EventType)

    var firstErr error
    for _, ch := range channels {
        payload := buildSendPayload(job, ch)

            if firstErr == nil { firstErr = err }
            slog.Error("notification.send", "channel", ch.Name(), "event", job.Args.EventType, "err", err)
            continue

    }

}
