```go
type ReviewerPool struct {
    Roles            []string
    Quorum           int      // resolved from asset.Quorum(); -1 → all
    RequireHuman     bool
    EscalationRoles  []string

}

type Resolver struct {
    db      *sql.DB
    yaml    *policy.YAMLConfig  // injected; may be nil
}

func (r *Resolver) ResolveReviewers(
    ctx context.Context,
    asset *asset.Asset,
    tags []string,        // tags currently on the asset's columns/asset
    owner string,         // asset_metadata.owner
) (ReviewerPool, error)
```
Body:
```go
pool := ReviewerPool{Quorum: int(asset.Quorum()), RequireHuman: asset.RequireHumanReview(), EscalationRoles: asset.EscalationRoles()}
if pool.Quorum == 0 { pool.Quorum = 1 }

// (1) Builder
if br := asset.ReviewerRoles(); len(br) > 0 {
    pool.Roles = append(pool.Roles, br...)
    pool.Source = append(pool.Source, "builder")
}

// (2) YAML tag rules
if r.yaml != nil {
    for _, tag := range tags {
        if roles, ok := r.yaml.TagReviewerRoles[tag]; ok {
            pool.Roles = append(pool.Roles, roles...)
            pool.Source = append(pool.Source, "yaml-tag:"+tag)
        }
    }
}

// (3) Owner fallback ONLY if (1)∪(2) empty
if len(pool.Roles) == 0 && owner != "" {
    var roles []string
    err := r.db.QueryRowContext(ctx, `SELECT roles FROM team_owners WHERE owner_email=$1`, owner).Scan(pq.Array(&roles))
    if err == nil && len(roles) > 0 {
        pool.Roles = roles
        pool.Source = append(pool.Source, "owner-fallback")
    }
}

return dedupRoles(pool), nil
```

    - **Submit**: validates code_hash matches latest asset_versions row; resolves reviewer pool; runs `AutoApprovalChecker.Check`. If `DecisionAutoApproved`: opens tx, transitions `governance_state='active'` directly, INSERTs `governance_reviews` row with `status='auto_approved'`, writes `governance.auto_approved` audit, enqueues notification job to owner. If `DecisionMustHumanReview` or `DecisionBlocked`: opens tx, transitions `governance_state='in_review'`, INSERTs row with `status='in_review'`, writes `governance.submitted` audit, enqueues notification job to reviewer pool. If `DecisionBlocked`: includes the block reason in payload + the audit entry; status still `in_review` so the reviewer can see and address.
    - **Approve / Reject**: requires permission `/governance/reviews/*` write. Approve: tx → UPDATE governance_reviews SET status, decided_at, decided_by_id, comment → UPDATE asset_versions SET governance_state='active' (or 'rejected') → audit.WriteEntry → enqueue notification to submitter → commit. Reject MUST require non-empty comment (400 if empty). Quorum logic: if quorum > 1 and quorum is not yet met by approve count, leave status='in_review' and just record the partial vote (extension table or in `comment` field — for v1 simplest: each Approve appends to comment "[role:X approved]" and only when count >= quorum the status flips).
    - **Reassign** (Pitfall #12): rotates the `reviewer_pool_snapshot.Roles` for an in-flight review. Adds a `governance.reviewer_reassigned` event to event_log (NOT audit_log — this is operational, not access-control mutation). Audit log captures via the next decision write.
    - **Status**: returns current state of all reviews for an asset (or all assets if no asset arg).
    - **Materialization gate** (D-08): in `executor.runStep`, BEFORE Resolve+Materialize, if `governance.gating_enabled` config flag is true:
      ```go
      var state string
      if err := db.QueryRowContext(ctx, `SELECT governance_state FROM asset_versions WHERE asset=$1 AND code_hash=$2`, a.Name(), a.CodeHash()).Scan(&state); err == nil && state != "active" {
          e.appendEvent(ctx, runID, event.EventTypeGovernanceMaterializationBlocked, ...)
          // Audit log (governance is access control)
          tx, _ := db.BeginTx(...); audit.WriteEntry(ctx, tx, audit.Entry{EventType: AuditGovernanceMaterializationBlocked, ...}); tx.Commit()
          return ErrMaterializationGated
      }
      ```
      `governance.gating_enabled` defaults to **false** for v1 (D-08).

For each `governance_reviews` row where `decided_at IS NULL AND submitted_at + interval $sla_hours hours < NOW() AND sla_breach_emitted_at IS NULL`: emit `governance.review_sla_breached` audit_log + enqueue notification to all reviewers + owner; set `sla_breach_emitted_at = NOW()` so we don't re-emit per tick.
