package api

import (
    "context"
    "errors"
    "net/http"
    "strings"
    "time"

    "github.com/kanpon/data-governance/internal/storage/ent"
    "github.com/kanpon/data-governance/internal/storage/ent/assetversion"
    "github.com/kanpon/data-governance/internal/storage/ent/columnege"
)

// SearchResult 代表单个搜索结果项。
type SearchResult struct {
    Type        string   `json:"type"` // "asset", "column"
    Name        string   `json:"name"`
    Description string   `json:"description"`
    Owner       string   `json:"owner"`
    Tags        []string `json:"tags"`
    AssetName   string   `json:"asset_name"` // 对于列：父资产
    Highlight   string   `json:"highlight"`
}

// SearchResponse 是 GET /v1/catalog/search 的响应。
type SearchResponse struct {
    Results  []SearchResult `json:"results"`
    Total    int            `json:"total"`
    Page     int            `json:"page"`
    Query    string         `json:"query"`
    Tags     []string       `json:"tags"`     // 所有可用于过滤碎片的标签
    Owners   []string       `json:"owners"`   // 所有可用于下拉的所有者
}

// searchHandler 处理 GET /v1/catalog/search。
// 查询参数：q（可选）、type（可选：asset|column|table）、tag（可选）、owner（可选）、page（可选，默认 1）。
// 当 q 为空但设置了 tag/owner 时，执行带过滤的浏览（UI-03）。
// 当 q 设置时，执行带过滤的 FTS 搜索（META-04）。
func searchHandler(deps Deps) http.HandlerFunc {
    return func(w http.ResponseWriter, r *http.Request) {
        query := strings.TrimSpace(r.URL.Query().Get("q"))
        tag := strings.TrimSpace(r.URL.Query().Get("tag"))
        owner := strings.TrimSpace(r.URL.Query().Get("owner"))
        searchType := r.URL.Query().Get("type") // asset, column, 或空（全部）
        page := 1
        if p := r.URL.Query().Get("page"); p != "" {
            if v := parsePositiveInt(p); v > 0 {
                page = v
            }
        }
        pageSize := 20
        offset := (page - 1) * pageSize

        ctx, cancel := context.WithTimeout(r.Context(), 5*time.Second)
        defer cancel()

        results, total, tags, owners, err := performSearch(ctx, deps, query, tag, owner, searchType, pageSize, offset)
        if err != nil {
            InternalServerError(w, err.Error())
            return
        }

        writeJSONResponse(w, http.StatusOK, SearchResponse{
            Results: results,
            Total:   total,
            Page:    page,
            Query:   query,
            Tags:    tags,
            Owners:  owners,
        })
    }
}

// performSearch 执行 FTS 查询，带可选的标签/所有者过滤器。
// 当 query 为空时：浏览模式（UI-03） - 返回匹配标签/所有者过滤的所有资产/列
// 当 query 设置时：搜索模式（META-04） - 应用标签/所有者过滤的 FTS 搜索
//
// FTS 查询模式（原始 pgx 或 ent 原始 SQL）：
//
// 资产子查询（用于 type=asset 或 type=all）：
//   SELECT name, description, owner, tags, NULL::text as asset_name,
//          ts_rank(search_vector, plainto_tsquery('english', $1)) as rank,
//          ts_headline('english', name || ' ' || coalesce(description, ''), plainto_tsquery('english', $1)) as highlight,
//          'asset' as type
//   FROM asset_versions
//   WHERE (search_vector @@ plainto_tsquery('english', $1) OR $1 = '')
//     AND superseded_at IS NULL
//     AND ($2 = '' OR tags @> ARRAY[$2])    -- 标签过滤器
//     AND ($3 = '' OR owner = $3)           -- 所有者过滤器
//
// 列子查询（用于 type=column 或 type=all，连接资产以获取 asset_name）：
//   SELECT col.name, col.description, col.owner, col.tags, a.name as asset_name,
//          ts_rank(col.search_vector, plainto_tsquery('english', $1)) as rank,
//          ts_headline('english', col.name || ' ' || coalesce(col.description, ''), plainto_tsquery('english', $1)) as highlight,
//          'column' as type
//   FROM column_edges col
//   JOIN asset_versions a ON col.asset_id = a.id
//   WHERE (col.search_vector @@ plainto_tsquery('english', $1) OR $1 = '')
//     AND col.superseded_at IS NULL
//     AND a.superseded_at IS NULL
//     AND ($2 = '' OR col.tags @> ARRAY[$2])
//     AND ($3 = '' OR col.owner = $3)
//
// 带 UNION ALL 的组合查询，按 rank DESC 排序，带 LIMIT/OFFSET：
//   SELECT * FROM (<asset_subquery> UNION ALL <column_subquery>) AS combined
//   ORDER BY rank DESC
//   LIMIT $4 OFFSET $5
//
// 总计数（无 LIMIT/OFFSET 的单独查询）：
//   SELECT COUNT(*) FROM (<asset_subquery> UNION ALL <column_subquery>) AS combined
//
// 带类型过滤器：类型不匹配时子查询包含 "AND false" 条件
//
// 标签聚合（用于过滤碎片）：
//   SELECT DISTINCT unnest(tags) FROM asset_versions WHERE superseded_at IS NULL ORDER BY 1
//
// 所有者聚合（用于下拉）：
//   SELECT DISTINCT owner FROM asset_versions WHERE superseded_at IS NULL AND owner != '' ORDER BY owner

func performSearch(ctx context.Context, deps Deps, query, tag, owner, searchType string, limit, offset int) ([]SearchResult, int, []string, []string, error) {
    // 验证搜索类型
    if searchType != "" && searchType != "asset" && searchType != "column" && searchType != "table" {
        return nil, 0, nil, nil, errors.New("invalid search type: must be empty, asset, column, or table")
    }

    // 使用原始 pgx 构建 FTS 查询（deps.Ent.DB() 提供 *sqlx.DB）
    // 确切 SQL 在上面的行为块中记录
    //
    // 关键过滤逻辑：
    // - $1 = 查询文本（浏览模式为空字符串，plainto_tsquery('english', '') 什么都不返回）
    // - $2 = 标签过滤器（空字符串表示无标签过滤）
    // - $3 = 所有者过滤器（空字符串表示无所有者过滤）
    // - 标签过滤器使用 PostgreSQL 数组包含：tags @> ARRAY[$2]
    // - 所有者过滤器使用精确匹配：owner = $3

    // 获取所有可用标签和所有者用于过滤 UI（UI-03）
    tags, err := fetchAvailableTags(ctx, deps)
    if err != nil {
        tags = []string{}
    }
    owners, err := fetchAvailableOwners(ctx, deps)
    if err != nil {
        owners = []string{}
    }

    // 占位符：返回 nil, 0, nil, nil 表示需要连接 FTS 查询
    // 实现将使用 ent 的 DB 连接（deps.Ent.DB()）与上述 SQL 模式
    // 使用 pgx 或 ent 的原始 SQL QueryContext。
    return nil, 0, tags, owners, errors.New("performSearch: FTS 查询将使用 ent 原始 SQL 与 deps.Ent.DB() 以及上述 tsvector/ts_rank/ts_headline 模式连接")
}

}

func fetchAvailableOwners(ctx context.Context, deps Deps) ([]string, error) {

    return nil, errors.New("fetchAvailableOwners: 将使用 ent 原始 SQL 连接")

func parsePositiveInt(s string) int {
    var v int
    for _, c := range s {
        if c < '0' || c > '9' {
            return 0
        }
        v = v*10 + int(c-'0')
    }
    return v

import { useState, useMemo } from 'react'
import { createFileRoute } from '@tanstack/react-router'
import { useQuery } from '@tanstack/react-query'
import { SearchBar } from '@/components/SearchBar'
import { SearchResult } from '@/components/SearchResult'
import { TagFilter } from '@/components/TagFilter'
import { OwnerSelect } from '@/components/OwnerSelect'
import { Spinner } from '@/components/ui/spinner'
import { Input } from '@/components/ui/input'

export function CatalogPage() {
  const [query, setQuery] = useState('')
  const [selectedTag, setSelectedTag] = useState('')
  const [selectedOwner, setSelectedOwner] = useState('')
  const [searchType, setSearchType] = useState<string>('') // '', 'asset', 'column'
  const [page, setPage] = useState(1)

  const { data, isLoading, isFetching } = useQuery({
    queryKey: ['catalog', 'search', query, selectedTag, selectedOwner, searchType, page],
    queryFn: async () => {
      const params = new URLSearchParams({
        page: String(page),
        page_size: '20',
      })
      if (query.trim()) params.set('q', query)
      if (selectedTag) params.set('tag', selectedTag)
      if (selectedOwner) params.set('owner', selectedOwner)
      if (searchType) params.set('type', searchType)
      const res = await fetch(`/v1/catalog/search?${params}`)
      if (!res.ok) throw new Error('Search failed')
      return res.json()
    },
    staleTime: 60 * 1000,
    refetchInterval: 60 * 1000, // D-17 冷屏
  })

  // 过滤器变化时清除页面
  const handleTagSelect = (tag: string) => {
    setSelectedTag(prev => prev === tag ? '' : tag)
    setPage(1)
  }

  const handleOwnerSelect = (owner: string) => {
    setSelectedOwner(owner)
    setPage(1)
  }

  const handleSearch = () => {
    setPage(1)
  }

  return (
    <div className="space-y-4 max-w-4xl mx-auto">
      <h1 className="text-2xl font-bold">数据目录</h1>

      {/* 搜索控制 */}
      <div className="space-y-3">
        <SearchBar
          value={query}
          onChange={setQuery}
          onSearch={handleSearch}
          placeholder="搜索资产、列、描述..."
        />

        {/* 标签过滤碎片（UI-03） */}
        {data?.tags?.length > 0 && (
          <TagFilter
            tags={data.tags}
            selectedTag={selectedTag}
            onSelect={handleTagSelect}
          />
        )}

        {/* 所有者过滤下拉（UI-03） */}
        {data?.owners?.length > 0 && (
          <OwnerSelect
            owners={data.owners}
            selectedOwner={selectedOwner}
            onSelect={handleOwnerSelect}
          />
        )}

        {/* 类型过滤器 */}
        <div className="flex items-center gap-2">
          <span className="text-sm text-muted-foreground">类型：</span>
          {['all', 'asset', 'column'].map(t => (
            <button
              key={t}
              onClick={() => { setSearchType(t === 'all' ? '' : t); setPage(1) }}
              className={`px-3 py-1 rounded text-sm ${
                (t === 'all' && !searchType) || searchType === t
                  ? 'bg-primary text-primary-foreground'
                  : 'bg-muted hover:bg-muted/80'
              }`}
            >
              {t === 'all' ? '全部' : t.charAt(0).toUpperCase() + t.slice(1)}
            </button>
          ))}
        </div>
      </div>

      {/* 结果 */}
      {isLoading && (
        <div className="flex justify-center py-12">
          <Spinner />
        </div>
      )}

      {data?.results?.length === 0 && !isLoading && (
        <div className="text-center py-12 text-muted-foreground">
          {query || selectedTag || selectedOwner
            ? `未找到当前过滤条件的结果`
            : '目录中尚无资产'}
        </div>
      )}

      {data?.results?.length > 0 && (
        <div className="space-y-3">
          <p className="text-sm text-muted-foreground">
            {data.total} 个结果
            {selectedTag && <> 标签为 <strong>"{selectedTag}"</strong></>}
            {selectedOwner && <> 所有者为 <strong>"{selectedOwner}"</strong></>}
            {query && <> 匹配 <strong>"{query}"</strong></>}
          </p>
          {data.results.map((result: any, i: number) => (
            <SearchResult
              key={`${result.type}-${result.name}-${i}`}
              result={result}
              query={query}
            />
          ))}
        </div>
      )}

      {/* 分页 */}
      {data?.total > 20 && (
        <div className="flex items-center justify-center gap-2 pt-4">
          <button
            onClick={() => setPage(p => p - 1)}
            disabled={page === 1}
            className="px-3 py-1 rounded border disabled:opacity-50"
          >
            上一页
          </button>
          <span className="text-sm">
            第 {page} 页，共 {Math.ceil(data.total / 20)} 页
          </span>
          <button
            onClick={() => setPage(p => p + 1)}
            disabled={page >= Math.ceil(data.total / 20)}
            className="px-3 py-1 rounded border disabled:opacity-50"
          >
            下一页
          </button>
        </div>
      )}
    </div>
  )
}
