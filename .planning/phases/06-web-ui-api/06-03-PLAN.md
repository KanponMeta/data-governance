package api

import (

// SearchResult represents a single search result item.
type SearchResult struct {
    Type        string `json:"type"` // "asset", "column"
    Name        string `json:"name"`
    Description string `json:"description"`
    Owner       string `json:"owner"`
    Tags        []string `json:"tags"`
    AssetName   string `json:"asset_name"` // for columns: parent asset
    Highlight   string `json:"highlight"`
}

// SearchResponse is GET /v1/catalog/search response.
type SearchResponse struct {
    Results []SearchResult `json:"results"`
    Total   int            `json:"total"`
    Page    int            `json:"page"`
    Query   string         `json:"query"`
}

// searchHandler handles GET /v1/catalog/search.
// Query params: q (required), type (optional: asset|column|table), page (optional, default 1).
func searchHandler(deps Deps) http.HandlerFunc {
    return func(w http.ResponseWriter, r *http.Request) {
        query := strings.TrimSpace(r.URL.Query().Get("q"))
        if query == "" {
            BadRequest(w, "q parameter is required")
            return
        }

        searchType := r.URL.Query().Get("type") // asset, column, or empty (all)
        page := 1
        if p := r.URL.Query().Get("page"); p != "" {
            if v := parsePositiveInt(p); v > 0 {
                page = v
            }
        }
        pageSize := 20
        offset := (page - 1) * pageSize

        ctx, cancel := context.WithTimeout(r.Context(), 5*time.Second)
        defer cancel()

        results, total, err := performSearch(ctx, deps, query, searchType, pageSize, offset)
        if err != nil {
            InternalServerError(w, err.Error())
            return
        }

        writeJSONResponse(w, http.StatusOK, SearchResponse{
            Results: results,
            Total:   total,
            Page:    page,
            Query:   query,
        })
    }
}

// performSearch executes FTS query against asset_versions and column_edges tables.
// Uses plainto_tsquery for simple free-text search with English stemmer.
// Results ordered by ts_rank DESC, weighted by type (asset higher than column).
//
// FTS Query Pattern (raw pgx or ent raw SQL):
//
// Asset subquery (for type=asset or type=all):
//   SELECT name, description, owner, tags, NULL::text as asset_name,
//          ts_rank(search_vector, plainto_tsquery('english', $1)) as rank,
//          ts_headline('english', name || ' ' || coalesce(description, ''), plainto_tsquery('english', $1)) as highlight,
//          'asset' as type
//   FROM asset_versions
//   WHERE search_vector @@ plainto_tsquery('english', $1)
//     AND superseded_at IS NULL
//
// Column subquery (for type=column or type=all, join with asset for asset_name):
//   SELECT col.name, col.description, col.owner, col.tags, a.name as asset_name,
//          ts_rank(col.search_vector, plainto_tsquery('english', $1)) as rank,
//          ts_headline('english', col.name || ' ' || coalesce(col.description, ''), plainto_tsquery('english', $1)) as highlight,
//          'column' as type
//   FROM column_edges col
//   JOIN asset_versions a ON col.asset_id = a.id
//   WHERE col.search_vector @@ plainto_tsquery('english', $1)
//     AND col.superseded_at IS NULL
//     AND a.superseded_at IS NULL
//
// Combined query with UNION ALL, ordered by rank DESC, with LIMIT/OFFSET:
//   SELECT * FROM (<asset_subquery> UNION ALL <column_subquery>) AS combined
//   ORDER BY rank DESC
//   LIMIT $2 OFFSET $3
//
// Total count (separate query without LIMIT/OFFSET):
//   SELECT COUNT(*) FROM (<asset_subquery> UNION ALL <column_subquery>) AS combined
//
// With type filter: subquery includes "AND false" condition when type doesn't match
//
// Implementation using ent raw SQL (example pattern):
//   err := deps.Ent.Debug().AssetVersion.Query().
//       WithDebug().
//       // Use raw SQL fallback for FTS since ent doesn't support tsvector ops natively
//       // Falls back to pgx raw query via deps.Ent.DB()
//
//   Or using sqlc-generated query with the raw SQL above wrapped as a stored procedure
//   or via ent's QueryContext with a raw SQL string.
//
// Key points:
// - plainto_tsquery('english', $1) handles user query as plain text (no operators)
// - ts_rank weights by frequency; search_vector weights already applied via setweight
// - ts_headline generates HTML highlight markers (<b>...</b>)
// - superseded_at IS NULL filters to current (non-deleted) versions
// - LIMIT/OFFSET for pagination
//
// Returns ([]SearchResult, totalCount, error).
func performSearch(ctx context.Context, deps Deps, query, searchType string, limit, offset int) ([]SearchResult, int, error) {
    // Validate search type
    if searchType != "" && searchType != "asset" && searchType != "column" && searchType != "table" {
        return nil, 0, errors.New("invalid search type: must be empty, asset, column, or table")
    }

    // Build the FTS query using raw pgx (deps.Ent.DB() provides *sqlx.DB)
    // The exact SQL is:
    //
    // -- Asset search subquery
    // SELECT
    //   av.name,
    //   av.description,
    //   av.owner,
    //   av.tags,
    //   NULL::text as asset_name,
    //   ts_rank(av.search_vector, plainto_tsquery('english', $1)) as rank,
    //   ts_headline('english', av.name || ' ' || coalesce(av.description, ''), plainto_tsquery('english', $1)) as highlight,
    //   'asset' as result_type
    // FROM asset_versions av
    // WHERE av.search_vector @@ plainto_tsquery('english', $1)
    //   AND av.superseded_at IS NULL
    //
    // -- Column search subquery
    // UNION ALL
    // SELECT
    //   ce.name,
    //   ce.description,
    //   ce.owner,
    //   ce.tags,
    //   a.name as asset_name,
    //   ts_rank(ce.search_vector, plainto_tsquery('english', $1)) as rank,
    //   ts_headline('english', ce.name || ' ' || coalesce(ce.description, ''), plainto_tsquery('english', $1)) as highlight,
    //   'column' as result_type
    // FROM column_edges ce
    // JOIN asset_versions a ON ce.asset_id = a.id
    // WHERE ce.search_vector @@ plainto_tsquery('english', $1)
    //   AND ce.superseded_at IS NULL
    //   AND a.superseded_at IS NULL
    //
    // Combine with type filter:
    // - type="asset": add "WHERE result_type = 'asset'" wrapper
    // - type="column"/"table": add "WHERE result_type = 'column'" wrapper
    //
    // Full query with limit/offset:
    // SELECT * FROM (<combined>) sub WHERE 1=1 [AND type_filter] ORDER BY rank DESC LIMIT $2 OFFSET $3

    // Placeholder: return nil, 0, nil to indicate FTS query wiring needed
    // The implementation wires ent's DB connection (deps.Ent.DB()) with the above SQL pattern
    // using pgx or ent's raw SQL QueryContext.
    return nil, 0, errors.New("performSearch: FTS query to be wired using ent raw SQL with deps.Ent.DB() and the tsvector/ts_rank/ts_headline pattern documented above")
}

func parsePositiveInt(s string) int {
    var v int
    for _, c := range s {
        if c < '0' || c > '9' {
            return 0
        }
        v = v*10 + int(c-'0')
    }
    return v
}

import { useState } from 'react'
import { createFileRoute } from '@tanstack/react-router'
import { useQuery } from '@tanstack/react-query'
import { SearchBar } from '@/components/SearchBar'
import { SearchResult } from '@/components/SearchResult'
import { Spinner } from '@/components/ui/spinner'
import { Input } from '@/components/ui/input'

export function CatalogPage() {
  const [query, setQuery] = useState('')
  const [searchType, setSearchType] = useState<string>('') // '', 'asset', 'column'
  const [page, setPage] = useState(1)

  const { data, isLoading, isFetching } = useQuery({
    queryKey: ['catalog', 'search', query, searchType, page],
    queryFn: async () => {
      if (!query.trim()) return null
      const params = new URLSearchParams({
        q: query,
        page: String(page),
        page_size: '20',
      })
      if (searchType) params.set('type', searchType)
      const res = await fetch(`/v1/catalog/search?${params}`)
      if (!res.ok) throw new Error('Search failed')
      return res.json()
    },
    enabled: query.trim().length > 0,
    staleTime: 60 * 1000,
    refetchInterval: 60 * 1000, // D-17 cold screen
  })

  return (
    <div className="space-y-4 max-w-4xl mx-auto">
      <h1 className="text-2xl font-bold">Catalog</h1>

      {/* Search controls */}
      <div className="space-y-2">
        <SearchBar
          value={query}
          onChange={setQuery}
          onSearch={() => setPage(1)}
          placeholder="Search assets, columns, tags, owners..."
        />
        <div className="flex items-center gap-2">
          <span className="text-sm text-muted-foreground">Filter:</span>
          {['all', 'asset', 'column'].map(t => (
            <button
              key={t}
              onClick={() => { setSearchType(t === 'all' ? '' : t); setPage(1) }}
              className={`px-3 py-1 rounded text-sm ${
                (t === 'all' && !searchType) || searchType === t
                  ? 'bg-primary text-primary-foreground'
                  : 'bg-muted hover:bg-muted/80'
              }`}
            >
              {t === 'all' ? 'All' : t.charAt(0).toUpperCase() + t.slice(1)}
            </button>
          ))}
        </div>
      </div>

      {/* Results */}
      {isLoading && (
        <div className="flex justify-center py-12">
          <Spinner />
        </div>
      )}

      {data?.results?.length === 0 && query && !isLoading && (
        <div className="text-center py-12 text-muted-foreground">
          No results found for "{query}"
        </div>
      )}

      {data?.results?.length > 0 && (
        <div className="space-y-3">
          <p className="text-sm text-muted-foreground">
            {data.total} result{data.total !== 1 ? 's' : ''} for "{query}"
          </p>
          {data.results.map((result: any, i: number) => (
            <SearchResult
              key={`${result.type}-${result.name}-${i}`}
              result={result}
              query={query}
            />
          ))}
        </div>
      )}

      {/* Pagination */}
      {data?.total > 20 && (
        <div className="flex items-center justify-center gap-2 pt-4">
          <button
            onClick={() => setPage(p => p - 1)}
            disabled={page === 1}
            className="px-3 py-1 rounded border disabled:opacity-50"
          >
            Previous
          </button>
          <span className="text-sm">
            Page {page} of {Math.ceil(data.total / 20)}
          </span>
          <button
            onClick={() => setPage(p => p + 1)}
            disabled={page >= Math.ceil(data.total / 20)}
            className="px-3 py-1 rounded border disabled:opacity-50"
          >
            Next
          </button>
        </div>
      )}
    </div>
  )
}

import { useState } from 'react'
import { Search } from 'lucide-react'
import { Input } from '@/components/ui/input'
import { Button } from '@/components/ui/button'

interface SearchBarProps {
  value: string
  onChange: (value: string) => void
  onSearch: () => void
  placeholder?: string
}

export function SearchBar({ value, onChange, onSearch, placeholder = 'Search...' }: SearchBarProps) {
  const handleKeyDown = (e: React.KeyboardEvent) => {
    if (e.key === 'Enter') {
      onSearch()
    }
  }

  return (
    <div className="flex gap-2">
      <div className="relative flex-1">
        <Search className="absolute left-3 top-1/2 -translate-y-1/2 text-muted-foreground" size={16} />
        <Input
          value={value}
          onChange={e => onChange(e.target.value)}
          onKeyDown={handleKeyDown}
          placeholder={placeholder}
          className="pl-9"
        />
      </div>
      <Button onClick={onSearch}>Search</Button>
    </div>
  )
}

import { useNavigate } from '@tanstack/react-router'
import { Card, CardContent } from '@/components/ui/card'
import { Badge } from '@/components/ui/badge'

interface SearchResultProps {
  result: {
    type: 'asset' | 'column'
    name: string
    description?: string
    owner?: string
    tags?: string[]
    asset_name?: string
    highlight?: string
  }
  query: string
}

export function SearchResult({ result, query }: SearchResultProps) {
  const navigate = useNavigate()

  const handleClick = () => {
    if (result.type === 'asset') {
      navigate({ to: `/assets/${result.name}` })
    } else if (result.asset_name) {
      navigate({ to: `/assets/${result.asset_name}` })
    }
  }

  // Highlight matched terms in the highlight field (from FTS ts_headline)
  const highlightHtml = result.highlight || result.description || ''

  return (
    <Card
      className="cursor-pointer hover:shadow-md transition-shadow"
      onClick={handleClick}
    >
      <CardContent className="p-4">
        <div className="flex items-start gap-3">
          <Badge variant={result.type === 'asset' ? 'default' : 'secondary'}>
            {result.type}
          </Badge>
          <div className="flex-1 min-w-0">
            <h3 className="font-semibold truncate">{result.name}</h3>
            {result.asset_name && result.type === 'column' && (
              <p className="text-xs text-muted-foreground">in {result.asset_name}</p>
            )}
            <p
              className="text-sm text-muted-foreground mt-1 line-clamp-2"
              dangerouslySetInnerHTML={{ __html: highlightHtml }}
            />
            <div className="flex items-center gap-2 mt-2">
              {result.owner && (
                <span className="text-xs text-muted-foreground">Owner: {result.owner}</span>
              )}
              {result.tags?.slice(0, 5).map(tag => (
                <Badge key={tag} variant="outline" className="text-xs">{tag}</Badge>
              ))}
            </div>
          </div>
        </div>
      </CardContent>
    </Card>
  )
}
