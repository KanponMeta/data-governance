// ListAssets returns all assets with pagination.
// GET /v1/assets?page=1&page_size=50
func listAssets(deps Deps) http.HandlerFunc {
    return func(w http.ResponseWriter, r *http.Request) {
        page := 1
        pageSize := 50
        if p := r.URL.Query().Get("page"); p != "" {
            if v, err := strconv.Atoi(p); err == nil && v > 0 {
                page = v
            }
        }
        if ps := r.URL.Query().Get("page_size"); ps != "" {
            if v, err := strconv.Atoi(ps); err == nil && v > 0 && v <= 100 {
                pageSize = v
            }
        }

        assets, total, err := listAssetsFromStorage(r.Context(), deps, page, pageSize)
        if err != nil {
            InternalServerError(w, err.Error())
            return
        }

        writeJSONResponse(w, http.StatusOK, map[string]any{
            "assets": assets,
            "total": total,
            "page": page,
        })
    }
}

import { createFileRoute } from '@tanstack/react-router'
import { useQuery, useQueryClient } from '@tanstack/react-query'
import { AssetCard } from '@/components/AssetCard'
import { Spinner } from '@/components/ui/spinner'
import { Button } from '@/components/ui/button'
import { Input } from '@/components/ui/input'

export function AssetDashboard() {
  const route = createFileRoute('/assets/')()
  const [search, setSearch] = useState('')
  const [page, setPage] = useState(1)

  // D-17: 60s polling for cold screens (asset dashboard)
  const { data, isLoading } = useQuery({
    queryKey: ['assets', page, search],
    queryFn: async () => {
      const params = new URLSearchParams({ page: String(page), page_size: '50' })
      if (search) params.set('q', search)
      const res = await fetch(`/v1/connect/api.v1.AssetService/ListAssets?${params}`)
      if (!res.ok) throw new Error('Failed to fetch assets')
      return res.json()
    },
    staleTime: 60 * 1000, // 60s (D-17 cold screen)
    refetchInterval: 60 * 1000, // poll every 60s
  })

  if (isLoading) {
    return (
      <div className="flex items-center justify-center h-64">
        <Spinner />
      </div>
    )
  }

  return (
    <div className="space-y-4">
      <div className="flex items-center gap-4">
        <h1 className="text-2xl font-bold">Assets</h1>
        <Input
          placeholder="Search assets..."
          value={search}
          onChange={e => setSearch(e.target.value)}
          className="max-w-xs"
        />
      </div>

      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
        {data?.assets?.map((asset: any) => (
          <AssetCard key={asset.name} asset={asset} />
        ))}
      </div>

      {data?.assets?.length === 0 && (
        <div className="text-center py-12 text-muted-foreground">
          No assets found.
        </div>
      )}

      {/* Pagination */}
      <div className="flex items-center justify-center gap-2">
        <Button variant="outline" onClick={() => setPage(p => p - 1)} disabled={page === 1}>
          Previous
        </Button>
        <span className="text-sm text-muted-foreground">Page {page} of {Math.ceil((data?.total || 0) / 50)}</span>
        <Button variant="outline" onClick={() => setPage(p => p + 1)} disabled={!data?.hasMore}>
          Next
        </Button>
      </div>
    </div>
  )
}

import { createFileRoute, useParams } from '@tanstack/react-router'
import { useQuery } from '@tanstack/react-query'
import { RunHistory } from '@/components/RunHistory'
import { QualityStatusBadge } from '@/components/QualityStatusBadge'
import { Card, CardHeader, CardTitle, CardContent } from '@/components/ui/card'
import { Badge } from '@/components/ui/badge'
import { Button } from '@/components/ui/button'
import { Spinner } from '@/components/ui/spinner'

export function AssetDetail() {
  const params = useParams({ from: '/assets/$name' })
  const assetName = params.name

  // Fetch asset metadata
  const { data: asset, isLoading: assetLoading } = useQuery({
    queryKey: ['assets', assetName],
    queryFn: async () => {
      const res = await fetch(`/v1/connect/api.v1.AssetService/GetAsset`, {
        headers: { 'Content-Type': 'application/json' },
      })
      if (!res.ok) throw new Error('Asset not found')
      return res.json()
    },
    staleTime: 30 * 1000,
  })

  // Fetch run history
  // D-17: hot screen (asset with active runs) = 3-5s polling
  // Determine polling interval based on latest run state
  const latestRunState = asset?.last_materialize_state
  const isHot = latestRunState === 'running' || latestRunState === 'queued'
  const pollingInterval = isHot ? 4 * 1000 : 60 * 1000 // 4s for hot, 60s for cold

  const { data: runsData, isLoading: runsLoading } = useQuery({
    queryKey: ['assets', assetName, 'runs'],
    queryFn: async () => {
      const res = await fetch(`/v1/connect/api.v1.AssetService/ListRuns?asset_name=${encodeURIComponent(assetName)}`)
      if (!res.ok) throw new Error('Failed to fetch runs')
      return res.json()
    },
    staleTime: 30 * 1000,
    refetchInterval: pollingInterval,
  })

  if (assetLoading || runsLoading) {
    return (
      <div className="flex items-center justify-center h-64">
        <Spinner />
      </div>
    )
  }

  return (
    <div className="space-y-6">
      {/* Asset header */}
      <div className="flex items-start justify-between">
        <div>
          <h1 className="text-2xl font-bold">{assetName}</h1>
          <div className="flex items-center gap-2 mt-1">
            <Badge variant={asset?.state === 'active' ? 'default' : 'secondary'}>
              {asset?.state}
            </Badge>
            <QualityStatusBadge state={asset?.last_materialize_state} />
          </div>
        </div>
        <Button variant="outline" onClick={() => history.back()}>Back</Button>
      </div>

      {/* Run history */}
      <div>
        <h2 className="text-lg font-semibold mb-3">Run History</h2>
        <RunHistory runs={runsData?.runs || []} />
      </div>
    </div>
  )
}

import { useState } from 'react'
import { RunStatusBadge } from './RunStatusBadge'
import { Card, CardContent, CardHeader } from '@/components/ui/card'
import { ChevronDown, ChevronRight } from 'lucide-react'

interface RunStep {
  name: string
  state: string
  started_at: string | null
  finished_at: string | null
  error: string | null
}

interface Run {
  id: string
  asset_name: string
  state: string
  queued_at: string
  started_at: string | null
  finished_at: string | null
  quality_state: string
  steps: RunStep[]
}

export function RunHistory({ runs }: { runs: Run[] }) {
  const [expandedRunId, setExpandedRunId] = useState<string | null>(null)

  if (runs.length === 0) {
    return <div className="text-muted-foreground py-8 text-center">No runs yet.</div>
  }

  return (
    <div className="space-y-3">
      {runs.map(run => (
        <div key={run.id}>
          {/* Run row */}
          <div
            className="flex items-center gap-3 p-3 border rounded-lg cursor-pointer hover:bg-muted/50"
            onClick={() => setExpandedRunId(expandedRunId === run.id ? null : run.id)}
          >
            {expandedRunId === run.id ? <ChevronDown size={16} /> : <ChevronRight size={16} />}
            <RunStatusBadge state={run.state} />
            <span className="font-mono text-sm">{run.id.slice(0, 8)}</span>
            <span className="text-sm text-muted-foreground">{run.asset_name}</span>
            <span className="text-sm text-muted-foreground ml-auto">
              {run.started_at ? new Date(run.started_at).toLocaleString() : '—'}
            </span>
          </div>

          {/* Expanded steps */}
          {expandedRunId === run.id && run.steps?.length > 0 && (
            <div className="ml-8 mt-2 space-y-2">
              {run.steps.map((step, i) => (
                <div key={i} className="flex items-center gap-3 p-2 bg-muted/30 rounded border">
                  <RunStatusBadge state={step.state} size="sm" />
                  <span className="text-sm font-medium">{step.name}</span>
                  <span className="text-xs text-muted-foreground ml-auto">
                    {step.started_at && new Date(step.started_at).toLocaleString()}
                    {step.error && <span className="text-destructive ml-2">{step.error}</span>}
                  </span>
                </div>
              ))}
            </div>
          )}
        </div>
      ))}
    </div>
  )
}
