```go
import (

    "connectrpc.com/connect"

type governanceConnectHandler struct {
    deps   ConnectDeps
    workflow *governance.Workflow
}

func newGovernanceConnectHandler(deps ConnectDeps) (string, *connect.Server) {
    h := &governanceConnectHandler{
        deps:     deps,
        workflow: deps.Workflow, // Phase 5 工作流服务
    }
    return "/v1/connect/api.v1.GovernanceService", connect.NewServer(h)
}

func (h *governanceConnectHandler) ListReviews(ctx context.Context, req *connect.Request[v1.ListReviewsRequest]) (*connect.Response[v1.ListReviewsResponse], error) {
    status := req.Msg.Status // "pending", "approved", "rejected", 或 ""（全部）
    page := int(req.Msg.Page)
    pageSize := int(req.Msg.PageSize)
    if pageSize <= 0 {
        pageSize = 20
    }

    // 从 ent/存储查询治理审查
    // reviews, total, err := h.workflow.ListReviews(ctx, status, page, pageSize)

    reviews := []*v1.ReviewResponse{}
    total := int32(0)

    // 占位符：返回空
    return connect.NewResponse(&v1.ListReviewsResponse{
        Reviews: reviews,
        Total:   total,
    }), nil
}

func (h *governanceConnectHandler) GetReview(ctx context.Context, req *connect.Request[v1.GetReviewRequest]) (*connect.Response[v1.ReviewResponse], error) {
    reviewID := req.Msg.Id
    if reviewID == "" {

        Id: reviewID,
    }), nil
}

        Id:     reviewID,
        Status: "approved",
    }), nil
}
```

```typescript
import { useState } from 'react'
import { createFileRoute } from '@tanstack/react-router'
import { useQuery, useQueryClient } from '@tanstack/react-query'
import { ReviewCard } from '@/components/ReviewCard'
import { ReviewModal } from '@/components/ReviewModal'
import { Spinner } from '@/components/ui/spinner'
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card'
import { Tabs, TabsContent, TabsList, TabsTrigger } from '@/components/ui/tabs'
import { Bell } from 'lucide-react'

interface Review {
  id: string
  asset_name: string
  status: 'pending' | 'approved' | 'rejected'
  submitted_by: string
  submitted_at: string
  reviewed_by?: string
  reviewed_at?: string
  comments: Array<{
    author: string
    body: string
    created_at: string
  }>
}

interface SessionInfo {
  permissions: {
    canApprove: boolean
  }
}

export function GovernancePage() {
  const [selectedReview, setSelectedReview] = useState<Review | null>(null)
  const [activeTab, setActiveTab] = useState<'pending' | 'approved' | 'rejected'>('pending')
  const queryClient = useQueryClient()

  // 获取会话以检查 canApprove 权限
  const { data: session } = useQuery<SessionInfo>({
    queryKey: ['me'],
    queryFn: async () => {
      const res = await fetch('/v1/me')
      if (!res.ok) throw new Error('Not authenticated')
      return res.json()
    },
    staleTime: 5 * 60 * 1000, // 5min - 会话很少变化
  })

  const canApprove = session?.permissions?.canApprove ?? false

  // D-17: 治理收件箱 15-30s 轮询（热屏）
  const { data: pendingData, isLoading: pendingLoading } = useQuery({
    queryKey: ['governance', 'reviews', 'pending'],
    queryFn: async () => {
      const res = await fetch('/v1/connect/api.v1.GovernanceService/ListReviews?status=pending')
      if (!res.ok) throw new Error('Failed to fetch reviews')
      return res.json()
    },
    enabled: canApprove,
    staleTime: 15 * 1000,
    refetchInterval: 20 * 1000, // 20s 轮询（D-17 热屏）
  })

  const { data: approvedData } = useQuery({
    queryKey: ['governance', 'reviews', 'approved'],
    queryFn: async () => {
      const res = await fetch('/v1/connect/api.v1.GovernanceService/ListReviews?status=approved')
      if (!res.ok) throw new Error('Failed to fetch reviews')
      return res.json()
    },
    enabled: canApprove,
    staleTime: 30 * 1000,
    refetchInterval: 60 * 1000,
  })

  const { data: rejectedData } = useQuery({
    queryKey: ['governance', 'reviews', 'rejected'],
    queryFn: async () => {
      const res = await fetch('/v1/connect/api.v1.GovernanceService/ListReviews?status=rejected')
      if (!res.ok) throw new Error('Failed to fetch reviews')
      return res.json()
    },
    enabled: canApprove,
    staleTime: 30 * 1000,
    refetchInterval: 60 * 1000,
  })

  const handleReviewAction = async () => {
    // 批准/拒绝后，关闭模态框并刷新
    setSelectedReview(null)
    queryClient.invalidateQueries({ queryKey: ['governance', 'reviews'] })
  }

  if (!canApprove) {
    return (
      <div className="flex items-center justify-center h-64">
        <Card className="max-w-md">
          <CardContent className="pt-6 text-center">
            <p className="text-muted-foreground">
              您没有权限访问治理收件箱。
            </p>
          </CardContent>
        </Card>
      </div>
    )
  }

  const pendingReviews: Review[] = pendingData?.reviews || []
  const approvedReviews: Review[] = approvedData?.reviews || []
  const rejectedReviews: Review[] = rejectedData?.reviews || []

  return (
    <div className="space-y-4">
      <div className="flex items-center gap-2">
        <Bell size={20} />
        <h1 className="text-2xl font-bold">治理收件箱</h1>
      </div>

      <Tabs value={activeTab} onValueChange={(v) => setActiveTab(v as typeof activeTab)}>
        <TabsList>
          <TabsTrigger value="pending">
            待处理
            {pendingReviews.length > 0 && (
              <span className="ml-2 px-1.5 py-0.5 rounded-full bg-destructive text-destructive-foreground text-xs">
                {pendingReviews.length}
              </span>
            )}
          </TabsTrigger>
          <TabsTrigger value="approved">已批准</TabsTrigger>
          <TabsTrigger value="rejected">已拒绝</TabsTrigger>
        </TabsList>

        <TabsContent value="pending" className="mt-4">
          {pendingLoading ? (
            <div className="flex justify-center py-12">
              <Spinner />
            </div>
          ) : pendingReviews.length === 0 ? (
            <Card>
              <CardContent className="pt-6 text-center text-muted-foreground">
                暂无待处理审查。
              </CardContent>
            </Card>
          ) : (
            <div className="space-y-3">
              {pendingReviews.map(review => (
                <ReviewCard
                  key={review.id}
                  review={review}
                  onClick={() => setSelectedReview(review)}
                />
              ))}
            </div>
          )}
        </TabsContent>

        <TabsContent value="approved" className="mt-4">
          <div className="space-y-3">
            {approvedReviews.map(review => (
              <ReviewCard key={review.id} review={review} onClick={() => setSelectedReview(review)} />
            ))}
            {approvedReviews.length === 0 && (
              <Card><CardContent className="pt-6 text-center text-muted-foreground">暂无已批准审查。</CardContent></Card>
            )}
          </div>
        </TabsContent>

        <TabsContent value="rejected" className="mt-4">
          <div className="space-y-3">
            {rejectedReviews.map(review => (
              <ReviewCard key={review.id} review={review} onClick={() => setSelectedReview(review)} />
            ))}
            {rejectedReviews.length === 0 && (
              <Card><CardContent className="pt-6 text-center text-muted-foreground">暂无已拒绝审查。</CardContent></Card>
            )}
          </div>
        </TabsContent>
      </Tabs>

      {/* 审查模态框 */}
      {selectedReview && (
        <ReviewModal
          review={selectedReview}
          onClose={() => setSelectedReview(null)}
          onAction={handleReviewAction}
        />
      )}
    </div>
  )
}
```

```typescript
import { useState } from 'react'
import { Dialog, DialogContent, DialogHeader, DialogTitle, DialogFooter } from '@/components/ui/dialog'
import { Button } from '@/components/ui/button'
import { Textarea } from '@/components/ui/textarea'
import { Label } from '@/components/ui/label'
import { Spinner } from '@/components/ui/spinner'

interface Review {
  id: string
  asset_name: string
  status: string
  submitted_by: string
  submitted_at: string
  comments: Array<{
    author: string
    body: string
    created_at: string
  }>
}

interface ReviewModalProps {
  review: Review
  onClose: () => void
  onAction: () => void
}

export function ReviewModal({ review, onClose, onAction }: ReviewModalProps) {
  const [comment, setComment] = useState('')
  const [isSubmitting, setIsSubmitting] = useState(false)
  const [error, setError] = useState<string | null>(null)

  const handleApprove = async () => {
    if (!comment.trim()) {
      setError('评论是必需的')
      return
    }
    await submitReview('approve')
  }

  const handleReject = async () => {
    if (!comment.trim()) {
      setError('评论是必需的')
      return
    }
    await submitReview('reject')
  }

  const submitReview = async (action: 'approve' | 'reject') => {
    setIsSubmitting(true)
    setError(null)

    try {
      // 从 cookie 获取 CSRF 令牌
      const csrfToken = document.cookie
        .split('; ')
        .find(row => row.startsWith('dg_csrf='))
        ?.split('=')[1] || ''

      const endpoint = action === 'approve'
        ? '/v1/connect/api.v1.GovernanceService/ApproveReview'
        : '/v1/connect/api.v1.GovernanceService/RejectReview'

      const res = await fetch(endpoint, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'X-CSRF-Token': csrfToken,
        },
        body: JSON.stringify({
          id: review.id,
          comment: comment,
        }),
      })

      if (!res.ok) {
        throw new Error(`Failed to ${action} review`)
      }

      onAction()
    } catch (err) {
      setError(err instanceof Error ? err.message : 'An error occurred')
    } finally {
      setIsSubmitting(false)
    }
  }

  return (
    <Dialog open onOpenChange={(open) => !open && onClose()}>
      <DialogContent className="max-w-lg">
        <DialogHeader>
          <DialogTitle>审查：{review.asset_name}</DialogTitle>
        </DialogHeader>

        <div className="space-y-4 py-4">
          {/* 审查详情 */}
          <div className="space-y-2 text-sm">
            <div className="flex justify-between">
              <span className="text-muted-foreground">提交者</span>
              <span>{review.submitted_by}</span>
            </div>
            <div className="flex justify-between">
              <span className="text-muted-foreground">提交时间</span>
              <span>{new Date(review.submitted_at).toLocaleString()}</span>
            </div>
            <div className="flex justify-between">
              <span className="text-muted-foreground">状态</span>
              <span className="capitalize">{review.status}</span>
            </div>
          </div>

          {/* 之前的评论 */}
          {review.comments.length > 0 && (
            <div className="border rounded-lg p-3 space-y-2">
              <p className="text-sm font-medium">评论</p>
              {review.comments.map((c, i) => (
                <div key={i} className="text-sm">
                  <span className="font-medium">{c.author}</span>
                  <span className="text-muted-foreground"> — {new Date(c.created_at).toLocaleString()}</span>
                  <p className="mt-0.5">{c.body}</p>
                </div>
              ))}
            </div>
          )}

          {/* 评论输入 */}
          <div className="space-y-2">
            <Label htmlFor="comment">评论 {review.status === 'pending' && '（必需）'}</Label>
            <Textarea
              id="comment"
              value={comment}
              onChange={(e) => setComment(e.target.value)}
              placeholder="输入您的审查评论..."
              rows={4}
            />
            {error && <p className="text-sm text-destructive">{error}</p>}
          </div>
        </div>

        <DialogFooter>
          <Button variant="outline" onClick={onClose} disabled={isSubmitting}>
            取消
          </Button>
          {review.status === 'pending' && (
            <>
              <Button variant="destructive" onClick={handleReject} disabled={isSubmitting}>
                {isSubmitting ? <Spinner size="sm" /> : '拒绝'}
              </Button>
              <Button onClick={handleApprove} disabled={isSubmitting}>
                {isSubmitting ? <Spinner size="sm" /> : '批准'}
              </Button>
            </>
          )}
        </DialogFooter>
      </DialogContent>
    </Dialog>
  )
}
```
