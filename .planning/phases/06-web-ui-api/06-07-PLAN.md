import (

type adminConnectHandler struct {
    deps ConnectDeps

func newAdminConnectHandler(deps ConnectDeps) (string, *connect.Server) {
    h := &adminConnectHandler{deps: deps}
    return "/v1/connect/api.v1.AdminService", connect.NewServer(h)
}

    }

func (h *adminConnectHandler) ListUsers(ctx context.Context, req *connect.Request[v1.ListUsersRequest]) (*connect.Response[v1.ListUsersResponse], error) {
    if err := checkAdminPermission(ctx, h.deps, "users"); err != nil {
        return nil, connect.NewError(connect.CodePermissionDenied, nil)
    }

    page := int(req.Msg.Page)
    pageSize := int(req.Msg.PageSize)
    if pageSize <= 0 {
        pageSize = 50
    }

    // 从 ent 查询用户：deps.Ent.User.Query().Offset(offset).Limit(pageSize).All(ctx)
    // 通过 ent 边包含角色

    total := int32(0)

    return connect.NewResponse(&v1.ListUsersResponse{
        Users: users,
        Total: total,
    }), nil
}

import { useState } from 'react'
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query'
import { UserTable } from '@/components/UserTable'
import { Button } from '@/components/ui/button'
import { Spinner } from '@/components/ui/spinner'

interface User {
  id: string
  email: string
  name: string
  roles: string[]
  last_active_at: string | null
}

export function UsersTab() {
  const queryClient = useQueryClient()
  const [page, setPage] = useState(1)

  const { data, isLoading } = useQuery({
    queryKey: ['admin', 'users', page],
    queryFn: async () => {
      const res = await fetch(`/v1/connect/api.v1.AdminService/ListUsers?page=${page}&page_size=20`)
      if (!res.ok) throw new Error('Failed to fetch users')
      return res.json()
    },
    staleTime: 30 * 1000,
  })

  const assignRole = useMutation({
    mutationFn: async ({ userId, role }: { userId: string; role: string }) => {
      const res = await fetch('/v1/connect/api.v1.AdminService/AssignRole', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ user_id: userId, role }),
      })
      if (!res.ok) throw new Error('Failed to assign role')
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['admin', 'users'] })
    },
  })

  const removeRole = useMutation({
    mutationFn: async ({ userId, role }: { userId: string; role: string }) => {
      const res = await fetch('/v1/connect/api.v1.AdminService/RemoveRole', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ user_id: userId, role }),
      })
      if (!res.ok) throw new Error('Failed to remove role')
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['admin', 'users'] })
    },
  })

  if (isLoading) {
    return <div className="flex justify-center py-12"><Spinner /></div>
  }

  return (
    <div className="space-y-4">
      <UserTable
        users={data?.users || []}
        onAssignRole={(userId, role) => assignRole.mutate({ userId, role })}
        onRemoveRole={(userId, role) => removeRole.mutate({ userId, role })}
      />

      {data?.total > 20 && (
        <div className="flex justify-center gap-2">
          <Button variant="outline" onClick={() => setPage(p => p - 1)} disabled={page === 1}>
            上一页
          </Button>
          <span className="py-2 text-sm">第 {page} 页</span>
          <Button variant="outline" onClick={() => setPage(p => p + 1)} disabled={!data?.hasMore}>
            下一页
          </Button>
        </div>
      )}
    </div>
  )
}

import { useState } from 'react'
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query'
import { Button } from '@/components/ui/button'
import { RoleForm } from '@/components/RoleForm'
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card'
import { Trash2 } from 'lucide-react'

interface Role {
  name: string
  description: string
  created_at: string
}

export function RolesTab() {
  const queryClient = useQueryClient()
  const [showCreate, setShowCreate] = useState(false)

  const { data: roles, isLoading } = useQuery({
    queryKey: ['admin', 'roles'],
    queryFn: async () => {
      const res = await fetch('/v1/connect/api.v1.AdminService/ListRoles')
      if (!res.ok) throw new Error('Failed to fetch roles')
      return res.json()
    },
    staleTime: 60 * 1000,
  })

  const createRole = useMutation({
    mutationFn: async ({ name, description }: { name: string; description: string }) => {
      const res = await fetch('/v1/connect/api.v1.AdminService/CreateRole', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ name, description }),
      })
      if (!res.ok) throw new Error('Failed to create role')
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['admin', 'roles'] })
      setShowCreate(false)
    },
  })

  const deleteRole = useMutation({
    mutationFn: async (name: string) => {
      const res = await fetch('/v1/connect/api.v1.AdminService/DeleteRole', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ name }),
      })
      if (!res.ok) throw new Error('Failed to delete role')
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['admin', 'roles'] })
    },
  })

  return (
    <div className="space-y-4">
      <div className="flex justify-between items-center">
        <h2 className="text-lg font-semibold">角色</h2>
        <Button onClick={() => setShowCreate(true)}>创建角色</Button>

        <Card>
          <CardHeader><CardTitle>创建角色</CardTitle></CardHeader>

              onSubmit={(name, description) => createRole.mutate({ name, description })}

              isLoading={createRole.isPending}
            />
          </CardContent>
        </Card>
      )}

      <div className="space-y-2">
        {roles?.roles?.map((role: Role) => (

            <CardContent className="p-4 flex items-center justify-between">
              <div>
                <p className="font-medium">{role.name}</p>
                <p className="text-sm text-muted-foreground">{role.description}</p>

                variant="ghost"
                size="icon"

                  }
                }}
              >

func main() {
    // ... 现有设置代码 ...

    // 构建 API deps（现有代码）
    deps := api.Deps{
        // ... 现有 deps ...
    }

    // 创建 chi 路由器
    r := api.NewRouter(deps)

    // 为非 API 路由包装 SPA 处理器
    // 生产中（从二进制运行时），提供嵌入的 SPA

    var handler http.Handler
    if isProductionBuild() {
        // 为非 API 路由提供嵌入的 React SPA
        spaHandler := spaFileServer(staticAssets)
        r = wrapWithSPA(r, spaHandler)
    } else {
        // 开发模式：仅 API，Vite 提供 SPA
        handler = r
    }

    // ... 现有服务器启动 ...
}

func isProductionBuild() bool {
    // 检查是否从嵌入二进制运行 vs 开发模式
    // 可以通过构建标签或环境变量控制
    return len(os.Args) > 0 && !strings.Contains(os.Args[0], "dev")
}

func spaFileServer(assets embed.FS) http.Handler {
    // 从嵌入的 FS 中提取 'dist' 目录
    dist, err := fs.Sub(assets, "dist")
    if err != nil {
        slog.Error("failed to extract embedded dist", "error", err)
        return http.NotFoundHandler()
    }
    return http.FileServer(http.FS(dist))
}

func wrapWithSPA(apiHandler, spaHandler http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        // API 路由由 chi 处理
        if strings.HasPrefix(r.URL.Path, "/v1") || strings.HasPrefix(r.URL.Path, "/health") || strings.HasPrefix(r.URL.Path, "/metrics") {
            apiHandler.ServeHTTP(w, r)
            return
        }
        // 所有其他路由提供 SPA（SPA 路由）
        spaHandler.ServeHTTP(w, r)
    })
}
