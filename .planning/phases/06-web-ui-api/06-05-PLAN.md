```go
// QualityTrendPoint 是质量趋势中的一个数据点。
type QualityTrendPoint struct {
    RunID       string  `json:"run_id"`
    FinishedAt  string  `json:"finished_at"`
    Score       float64 `json:"score"` // 0-100
    State       string  `json:"state"`
    RuleResults []RuleResult `json:"rule_results"`
}

// RuleResult 是一个规则评估结果。
type RuleResult struct {
    RuleName   string `json:"rule_name"`
    Passed     bool   `json:"passed"`
    Value      string `json:"value"`
    Threshold  string `json:"threshold"`
}

// QualityTrendResponse 是 GET /v1/quality/trend 的响应。
type QualityTrendResponse struct {
    Asset  string             `json:"asset"`
    Points []QualityTrendPoint `json:"points"`
    AvgScore float64          `json:"avg_score"`
}

// qualityTrendHandler 返回资产的质量趋势。
// 查询参数：asset（必需），runs（可选，默认 30）。
func qualityTrendHandler(deps Deps) http.HandlerFunc {
    return func(w http.ResponseWriter, r *http.Request) {
        asset := r.URL.Query().Get("asset")
        if asset == "" {
            BadRequest(w, "asset parameter is required")
            return
        }

        runs := 30
        if r := r.URL.Query().Get("runs"); r != "" {
            if v, err := strconv.Atoi(r); err == nil && v > 0 && v <= 100 {
                runs = v
            }
        }

        ctx, cancel := context.WithTimeout(r.Context(), 10*time.Second)
        defer cancel()

        points, avgScore, err := fetchQualityTrend(ctx, deps, asset, runs)
        if err != nil {
            InternalServerError(w, err.Error())
            return
        }

        writeJSONResponse(w, http.StatusOK, QualityTrendResponse{
            Asset:    asset,
            Points:   points,
            AvgScore: avgScore,
        })
    }
}
```

```go
package api

import (
    "context"
    "connectrpc.com/connect"
    "github.com/kanpon/data-governance/proto/api/v1"
)

type qualityConnectHandler struct{}

func newQualityConnectHandler(deps ConnectDeps) (string, *connect.Server) {
    h := &qualityConnectHandler{}
    return "/v1/connect/api.v1.QualityService", connect.NewServer(h)
}

func (h *qualityConnectHandler) QualityTrend(ctx context.Context, req *connect.Request[v1.QualityTrendRequest]) (*connect.Response[v1.QualityTrendResponse], error) {
    // 查询运行以获取质量趋势
    // 返回 proto QualityTrendResponse
    return connect.NewResponse(&v1.QualityTrendResponse{
        Points: []*v1.QualityTrendPoint{},
    }), nil
}

func (h *qualityConnectHandler) ListAlerts(ctx context.Context, req *connect.Request[v1.ListAlertsRequest]) (*connect.Response[v1.ListAlertsResponse], error) {
    // 查询质量警报
    return connect.NewResponse(&v1.ListAlertsResponse{
        Alerts: []*v1.QualityAlert{},
    }), nil
}

func (h *qualityConnectHandler) AcknowledgeAlert(ctx context.Context, req *connect.Request[v1.AcknowledgeAlertRequest]) (*connect.Response[v1.AcknowledgeAlertResponse], error) {
    // 更新警报确认标志
    return connect.NewResponse(&v1.AcknowledgeAlertResponse{
        Acknowledged: true,
    }), nil
}
```

```typescript
import { useMemo } from 'react'
import {
  LineChart,
  Line,
  XAxis,
  YAxis,
  CartesianGrid,
  Tooltip,
  ResponsiveContainer,
  ReferenceLine,
  Dot,
} from 'recharts'
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card'

interface QualityTrendPoint {
  run_id: string
  finished_at: string
  score: number
  state: string
  rule_results?: Array<{
    rule_name: string
    passed: boolean
    value: string
    threshold: string
  }>
}

interface QualityTrendChartProps {
  assetName: string
  data: QualityTrendPoint[] | null
  isLoading: boolean
}

export function QualityTrendChart({ assetName, data, isLoading }: QualityTrendChartProps) {
  const chartData = useMemo(() => {
    if (!data) return []
    return data.map(point => ({
      run: point.run_id.slice(0, 8),
      score: point.score,
      state: point.state,
      date: new Date(point.finished_at).toLocaleDateString(),
      fullDate: new Date(point.finished_at).toLocaleString(),
      ruleResults: point.rule_results || [],
    }))
  }, [data])

  const avgScore = useMemo(() => {
    if (!data || data.length === 0) return 0
    return Math.round(data.reduce((sum, p) => sum + p.score, 0) / data.length)
  }, [data])

  const stateToColor = (state: string) => {
    switch (state) {
      case 'success': return '#22c55e' // 绿色
      case 'failed': return '#ef4444' // 红色
      case 'quality_failed': return '#f97316' // 橙色
      default: return '#94a3b8' // 灰色
    }
  }

  const CustomDot = (props: any) => {
    const { cx, cy, payload } = props
    if (!cx || !cy) return null
    return (
      <circle
        cx={cx}
        cy={cy}
        r={4}
        fill={stateToColor(payload.state)}
        stroke="#fff"
        strokeWidth={2}
      />
    )
  }

  const CustomTooltip = ({ active, payload }: any) => {
    if (!active || !payload?.length) return null
    const d = payload[0].payload
    return (
      <div className="bg-background border rounded-lg p-3 shadow-lg text-sm">
        <p className="font-medium">运行 {d.run}</p>
        <p className="text-muted-foreground">{d.fullDate}</p>
        <p className="mt-1">
          分数：<span className="font-bold" style={{ color: stateToColor(d.state) }}>{d.score}</span>/100
        </p>
        <p>状态：{d.state}</p>
        {d.ruleResults.length > 0 && (
          <div className="mt-2 border-t pt-2">
            <p className="font-medium mb-1">规则结果：</p>
            {d.ruleResults.map((r: any, i: number) => (
              <div key={i} className="flex items-center gap-2 text-xs">
                <span className={r.passed ? 'text-green-600' : 'text-red-600'}>
                  {r.passed ? '通过' : '失败'}
                </span>
                <span>{r.rule_name}</span>
                <span className="text-muted-foreground">
                  {r.value} {r.threshold && `/ ${r.threshold}`}
                </span>
              </div>
            ))}
          </div>
        )}
      </div>
    )
  }

  if (isLoading) {
    return (
      <Card>
        <CardHeader><CardTitle>质量趋势</CardTitle></CardHeader>
        <CardContent>
          <div className="h-64 flex items-center justify-center text-muted-foreground">
            加载中...
          </div>
        </CardContent>
      </Card>
    )
  }

  return (
    <Card>
      <CardHeader>
        <div className="flex items-center justify-between">
          <CardTitle>质量趋势：{assetName}</CardTitle>
          {avgScore > 0 && (
            <div className="text-sm">
              平均：<span className="font-bold">{avgScore}</span>/100
            </div>
          )}
        </div>
      </CardHeader>
      <CardContent>
        {chartData.length === 0 ? (
          <div className="h-64 flex items-center justify-center text-muted-foreground">
            暂无质量数据。
          </div>
        ) : (
          <ResponsiveContainer width="100%" height={300}>
            <LineChart data={chartData} margin={{ top: 5, right: 20, bottom: 5, left: 0 }}>
              <CartesianGrid strokeDasharray="3 3" stroke="#f0f0f0" />
              <XAxis
                dataKey="date"
                tick={{ fontSize: 12 }}
                tickLine={{ stroke: '#e0e0e0' }}
              />
              <YAxis
                domain={[0, 100]}
                tick={{ fontSize: 12 }}
                tickLine={{ stroke: '#e0e0e0' }}
                label={{ value: '分数', angle: -90, position: 'insideLeft', fontSize: 12 }}
              />
              <Tooltip content={<CustomTooltip />} />
              <ReferenceLine y={100} stroke="#22c55e" strokeDasharray="3 3" label={{ value: '目标', position: 'right', fontSize: 10 }} />
              <Line
                type="monotone"
                dataKey="score"
                stroke="#6366f1"
                strokeWidth={2}
                dot={<CustomDot />}
                activeDot={{ r: 6, stroke: '#6366f1', strokeWidth: 2 }}
              />
            </LineChart>
          </ResponsiveContainer>
        )}
      </CardContent>
    </Card>
  )
}
```

```typescript
import { useQuery } from '@tanstack/react-query'
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card'
import { Badge } from '@/components/ui/badge'
import { Button } from '@/components/ui/button'
import { Bell, X } from 'lucide-react'

interface QualityAlert {
  id: string
  asset_name: string
  rule_name: string
  severity: 'critical' | 'warning' | 'info'
  message: string
  created_at: string
  acknowledged: boolean
}

export function AlertList() {
  // D-17: 质量警报 15-30s 轮询（热屏）
  const { data: alerts, isLoading } = useQuery<QualityAlert[]>({
    queryKey: ['quality', 'alerts'],
    queryFn: async () => {
      const res = await fetch('/v1/quality/alerts')
      if (!res.ok) throw new Error('Failed to fetch alerts')
      return res.json()
    },
    staleTime: 15 * 1000,
    refetchInterval: 20 * 1000, // 20s 轮询（D-17 热屏）
  })

  const acknowledgeAlert = async (id: string) => {
    await fetch(`/v1/quality/alerts/${id}/acknowledge`, { method: 'POST' })
    // 查询将自动刷新
  }

  const severityVariant = {
    critical: 'destructive' as const,
    warning: 'secondary' as const,
    info: 'outline' as const,
  }

  const unacknowledged = alerts?.filter(a => !a.acknowledged) || []
  const acknowledged = alerts?.filter(a => a.acknowledged) || []

  return (
    <Card>
      <CardHeader>
        <div className="flex items-center justify-between">
          <CardTitle className="flex items-center gap-2">
            <Bell size={18} />
            质量警报
          </CardTitle>
          <Badge variant="outline">{unacknowledged.length} 个活动</Badge>
        </div>
      </CardHeader>
      <CardContent>
        {isLoading && <div className="text-center py-8 text-muted-foreground">加载中...</div>}

        {unacknowledged.length === 0 && acknowledged.length === 0 && (
          <div className="text-center py-8 text-muted-foreground">暂无质量警报。</div>
        )}

        {unacknowledged.length > 0 && (
          <div className="space-y-3 mb-4">
            <p className="text-sm font-medium text-destructive">活动</p>
            {unacknowledged.map(alert => (
              <div key={alert.id} className="flex items-start gap-3 p-3 border rounded-lg border-destructive/50 bg-destructive/5">
                <Badge variant={severityVariant[alert.severity]}>{alert.severity}</Badge>
                <div className="flex-1 min-w-0">
                  <p className="font-medium text-sm">{alert.asset_name}: {alert.rule_name}</p>
                  <p className="text-xs text-muted-foreground mt-0.5">{alert.message}</p>
                  <p className="text-xs text-muted-foreground mt-1">
                    {new Date(alert.created_at).toLocaleString()}
                  </p>
                </div>
                <Button variant="ghost" size="icon" onClick={() => acknowledgeAlert(alert.id)}>
                  <X size={14} />
                </Button>
              </div>
            ))}
          </div>
        )}

        {acknowledged.length > 0 && (
          <div className="space-y-3 opacity-60">
            <p className="text-sm font-medium text-muted-foreground">已确认</p>
            {acknowledged.slice(0, 5).map(alert => (
              <div key={alert.id} className="flex items-center gap-3 p-3 border rounded-lg bg-muted/30">
                <Badge variant={severityVariant[alert.severity]}>{alert.severity}</Badge>
                <div className="flex-1 min-w-0">
                  <p className="font-medium text-sm">{alert.asset_name}: {alert.rule_name}</p>
                </div>
                <span className="text-xs text-muted-foreground">
                  {new Date(alert.created_at).toLocaleDateString()}
                </span>
              </div>
            ))}
          </div>
        )}
      </CardContent>
    </Card>
  )
}
```
