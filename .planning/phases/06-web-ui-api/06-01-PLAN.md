```go
package api

import (
    "context"
    "net/http"

    "connectrpc.com/connect"
    "github.com/kanpon/data-governance/internal/auth"
    "github.com/kanpon/data-governance/internal/event"
    "github.com/kanpon/data-governance/internal/storage/ent"

    "github.com/kanpon/data-governance/proto/api/v1/apiv1connect"
)

// ConnectDeps 持有 ConnectRPC 处理器的依赖项。
// 镜像 api.Deps 但没有 chi 特定字段。
type ConnectDeps struct {
    AuthService   AuthServiceServer
    AssetService  AssetServiceServer
    LineageService LineageServiceServer
    GovernanceService GovernanceServiceServer
    AuthMW        func(http.Handler) http.Handler
    Enforcer      any // casbin.Enforcer
    Issuer        *auth.TokenIssuer
    Events        event.Writer
    Ent           *ent.Client
}

// AssetServiceServer 实现存根 — 实际逻辑在后续计划中从 chi 处理器迁移。
// 对于此计划：在 chi 路由器中连接 ConnectRPC 处理器。

func (s *Server) mountConnectRPC(deps ConnectDeps, r chi.Router) {
    // 在 /v1/connect 路径挂载 ConnectRPC 处理器
    // 所有 ConnectRPC 方法都有服务名前缀：/api.v1.AssetService/ListAssets
    assetPath, handler := newAssetConnectHandler(deps)
    lineagePath, lineageHandler := newLineageConnectHandler(deps)
    governancePath, governanceHandler := newGovernanceConnectHandler(deps)

    r.Mount(assetPath, handler)
    r.Mount(lineagePath, lineageHandler)
    r.Mount(governancePath, governanceHandler)
}
```

```go
package api

import (

// meHandler 返回 UI 的已认证用户信息。
// GET /v1/me
func meHandler(deps Deps) http.HandlerFunc {
    return func(w http.ResponseWriter, r *http.Request) {
        p, ok := auth.PrincipalFromContext(r.Context())
        if !ok {
            Unauthorized(w, "authentication required")
            return
        }

        info, err := deps.Auth.GetSessionInfo(r.Context(), p.UserID)
        if err != nil {
            InternalServerError(w, err.Error())
            return
        }

        writeJSONResponse(w, http.StatusOK, info)
    }
}
```

## 威胁模型

### 信任边界

| 边界 | 描述 |
|----------|-------------|
| browser → API | 不受信任的输入在此交叉；cookie 是 httpOnly，CSRF 令牌验证状态变更请求 |

| SPA → API | 同源由 Vite 代理强制；cookie 作用域限制为平台来源 |

| 威胁 ID | 类别 | 组件 | 处理方式 | 缓解计划 |
